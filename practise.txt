DROP FUNCTION IF EXISTS F1 //
CREATE FUNCTION F1 (empno INT)
  RETURNS INT
  READS SQL DATA
  DETERMINISTIC
BEGIN
 DECLARE experience INT;

  SELECT TIMESTAMPDIFF(YEAR, e.hire_date, CURDATE()) INTO experience
    FROM employees e WHERE e.employee_id = empno;

  RETURN experience;
END //
